function buildScheduleLabel(schedule: DailySchedule) {
  const blocks = schedule.blocks.map((timeBlock) =>
    getFormattedTimeRange(undefined, [timeBlock.start, timeBlock.end])
  );
  return blocks.join(', ');
}
