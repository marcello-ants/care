export function careDateToGQLFormat(careDate: CareDate): GQLCareDate {
  const careDates = {
    [CareDate.RIGHT_NOW]: GQLCareDate.RIGHT_NOW,
    [CareDate.WITHIN_A_WEEK]: GQLCareDate.WITHIN_A_WEEK,
    [CareDate.IN_1_2_MONTHS]: GQLCareDate.WITHIN_A_MONTH,
    [CareDate.JUST_BROWSING]: GQLCareDate.JUST_BROWSING,
  };
 
  return careDates[careDate];
